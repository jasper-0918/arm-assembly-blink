# Project Name
TARGET = led-blinker

# Toolchain
PREFIX = arm-none-eabi-
CC = $(PREFIX)gcc
LD = $(PREFIX)gcc
OBJCOPY = $(PREFIX)objcopy
SIZE = $(PREFIX)size

# Files
SOURCES = blink.s
OBJECTS = $(SOURCES:.s=.o)

# Compiler flags
CFLAGS = -mcpu=cortex-m4 -mthumb -nostdlib -ffreestanding
CFLAGS += -Wall -Wextra -g

# Linker flags
LDFLAGS = -T STM32F446RE.ld -nostdlib -static

# Build rules
all: $(TARGET).elf $(TARGET).bin

# Assemble .s files to .o
%.o: %.s
	$(CC) $(CFLAGS) -c $< -o $@

# Link
$(TARGET).elf: $(OBJECTS) STM32F446RE.ld
	$(LD) $(LDFLAGS) $(OBJECTS) -o $@
	$(SIZE) $@

# Generate binary for flashing
$(TARGET).bin: $(TARGET).elf
	$(OBJCOPY) -O binary $< $@

# Flash to board
flash: $(TARGET).bin
	st-flash write $(TARGET).bin 0x08000000

# Clean build artifacts
clean:
	rm -f $(OBJECTS) $(TARGET).elf $(TARGET).bin

# Debug target (starts GDB server)
debug: $(TARGET).elf
	st-util &
	arm-none-eabi-gdb $(TARGET).elf -ex "target remote :4242"

.PHONY: all flash clean debug
